## Coding

int a = 5;

e = c && (a > 2 ? (b == (a = a * 2)) : d);
j += ((h < ((f - g == 3) ? 3 : 2)) && (i < 5)) ? 1 : 2;

System.out.println("e: " + e);
System.out.println("j: " + j);
