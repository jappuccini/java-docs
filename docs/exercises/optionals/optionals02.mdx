        <<record>>

    class TelephoneBook {
        <<record>>
        entries: Map~Person, TelephoneNumber~
        +addEntry(person: Person, telephoneNumber: TelephoneNumber) void
        +getTelephoneNumberByName(name: String) Optional~TelephoneNumber~
    }
